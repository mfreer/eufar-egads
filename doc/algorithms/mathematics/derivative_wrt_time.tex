\index{derivative\_wrt\_time}
\algdesc{Time Derivative
}
{ %%%%%% Algorithm name %%%%%%
derivative\_wrt\_time
}
{ %%%%%% Algorithm summary %%%%%%
Calculation of the first time derivative of a generic parameter.
}
{ %%%%%% Category %%%%%%
Mathematics
}
{ %%%%%% Inputs %%%%%%
$x$ & Vector & Parameter to calculate first derivative \\
$t$ & Vector & Time signal [hhmmss] 
}
{ %%%%%% Outputs %%%%%%
$\dot{x]$ & Vector & First derivative of x [units of x / sec]
}
{ %%%%%% Formula %%%%%%
\begin{displaymath}
\dot{x} = \frac{x_{i+1} - x_{i-1}}{t_{i+1} - t_{i-1}}
\end{displaymath}

}
